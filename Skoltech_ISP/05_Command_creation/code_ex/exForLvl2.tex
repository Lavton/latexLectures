\newcommand{\allFigures}[1]{
    \newcount\myNum%
    \myNum=1% 
    \foreach \capt in {#1}{
    \expandafter\oneFigure\capt{\the\myNum}\relax
    \global\advance\myNum by 1\relax
    } 
}

    num: #3, lines: #1, caption: #2\\
}
