\documentclass{article}

% Define a new insertion register
\newinsert\myinsert

% Set parameters for the insert
\skip\myinsert=0pt      % No extra vertical space
\count\myinsert=1000    % Maximum number of inserts per page
\dimen\myinsert=8in     % Maximum height of inserts per page

% Define a command to use the insert
\newcommand{\myfootnote}[1]{%
  \insert\myinsert{%
    % Material to be inserted at the bottom of the page
    \footnotesize
    \interlinepenalty100 % Prevents page breaks within the insert
    \noindent\hbox to \hsize{\hfil\footnotemark}%
    \footnotetext{#1}%
  }%
  % Place the footnote mark in the text
  \footnotemark
}
